import MediaLink, { createIcon } from './MediaLink';

### Code:

```javascript
import MediaLink from '@department-of-veterans-affairs/formation-react/MediaLink';

// Minimal example
<MediaLink
  href="https://www.youtube.com/watch?v=xE0LzeRgdAk"
  title="VA disability compensation"
/>

/* Renders as
<a
  href="https://www.youtube.com/watch?v=xE0LzeRgdAk"
  target="_blank"
  rel="noopener noreferrer"
>
  <i
    role="img"
    aria-hidden="true"
    class="vads-u-padding-right--1 fab fa-youtube"
  />
  View video about VA disability compensation
  <dfn>on YouTube</dfn>
</a>
*/

// Internal video with no icon & custom content
<MediaLink
  href="https://www.va.gov/media/12345"
  icon={null}
  external={false}
  content={() => (
    <>
      Watch this <em>awesome</em> video about our new service!
    </>
  )}
/>

/* Renders as
<a href="https://www.va.gov/media/12345">
  Watch this <em>awesome</em> video about our new service!
</a>
*/

// Unsupported site
<MediaLink
  href="https://vimeo.com/279619945"
  icon="fab fa-vimeo"
  content={() => (
    <>
      Watch this video on why accessibility matters on <dfn>Vimeo</dfn>
    </>
  )}
/>

/* Renders as
<a
  href="https://vimeo.com/279619945"
  target="_blank"
  rel="noopener noreferrer"
>
  <i
    role="img"
    aria-hidden="true"
    class="vads-u-padding-right--1 fab fa-vimeo"
  />
  Watch this video on why accessibility matters on <dfn>Vimeo</dfn>
</a>
*/

// Showing all properties
<MediaLink
  href="https://www.va.gov/some_video_page"
  title="VA Information on something"

  // equivalent to `icon="fas fa-play-circle foo"`
  icon={createIcon('fas fa-play-circle foo')}
  content={({ props }) => <>View video about {props.title}</>}
  external={false}

  // extra properties to add to the link (not restricted)
  className="internal-video"
  data-test="bar"
  onClick={() => {}}
/>

/* Renders as
<a href="https://www.va.gov/some_video_page" class="internal-video">
  <i
    role="img"
    aria-hidden="true"
    class="vads-u-padding-right--1 fas fa-play-circle foo"
  />
  View video about VA Information on something
</a>
*/
```

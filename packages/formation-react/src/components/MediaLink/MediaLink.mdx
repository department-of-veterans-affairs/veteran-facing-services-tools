import MediaLink from './MediaLink';

### Code:

```javascript
import MediaLink from '@department-of-veterans-affairs/formation-react/MediaLink';

// Minimal example
<MediaLink
  href="https://www.youtube.com/watch?v=xE0LzeRgdAk"
  title="VA disability compensation"
/>

/* Renders as
<a
  href="https://www.youtube.com/watch?v=xE0LzeRgdAk"
  target="_blank"
  rel="noopener noreferrer"
>
  <i role="img" aria-hidden="true" class="vads-u-padding-right--1 fab fa-youtube"/>
  View video about VA disability compensation
  <dfn>on YouTube</dfn>
</a>
*/

// Unsupported site
<MediaLink
  href="https://vimeo.com/279619945"
  title="Why Accessibility Matters"
>
  <i role="img" aria-hidden="true" class="vads-u-padding-right--1 fab fa-youtube"/>
  Watch this video on why accessibility matters on <dfn>Vimeo</dfn>
</MediaLink>

/* Renders as
<a
  href="https://vimeo.com/279619945"\
  target="_blank"
  rel="noopener noreferrer"
>
  <i role="img" aria-hidden="true" class="vads-u-padding-right--1 fab fa-youtube" />
  Watch this video on why accessibility matters on <dfn>Vimeo</dfn>
</a>
*/

// Showing all properties
<MediaLink
  href="https://www.va.gov/some_video_page"
  title="VA Information about something"

  icon="fas fa-play-circle foo"
  template="View video about {title}"
  external={false}

  // any extra properties to add to the link
  className="internal-video"
  data-test="bar"
  onClick={() => {}}
/>

/* Renders as
<a href="https://www.va.gov/some_video_page" class="internal-video">
  <i
    role="img"
    aria-hidden="true"
    class="vads-u-padding-right--1 fas fa-play-circle foo"
  />
  View video about VA Information about something
</a>
*/

// Internal video with custom template
<MediaLink
  href="https://www.youtube.com/watch?v=12345"
  title={
    <>
      Check out this <em>awesome</em> <dfn>YouTube</dfn> video about our
      new product!
    </>
  }
  template="{title}"
/>

/* Renders as
<a
  href="https://www.youtube.com/watch?v=12345"
  target="_blank"
  rel="noopener noreferrer"
>
  <i role="img" aria-hidden="true" class="vads-u-padding-right--1 fab fa-youtube" />
  Check out this <em>awesome</em> <dfn>YouTube</dfn> video about our new product!
</a>
*/
```
